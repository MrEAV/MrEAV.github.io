⚠ Important: Put a 1kΩ resistor on the ESP32 TX → DFPlayer RX line. The DFPlayer RX pin is sensitive and the resistor prevents damage. The TX line from DFPlayer back to ESP32 doesn't need one.
✓ Pull-down resistors: Each button needs a 10kΩ resistor between its signal pin and GND. This ensures the pin reads LOW when the button isn't pressed. Without it, the pin floats and you get phantom button presses.

// STARTER CODE — ARDUINO IDE (ESP32)
Libraries needed: Install via Arduino Library Manager — "DFRobotDFPlayerMini" and "ESPAsyncWebServer" + "AsyncTCP"

1. Board Setup

In Arduino IDE: Tools → Board → ESP32 Arduino → ESP32 Dev Module. Tools → Port → your COM port.

2. Main Firmware

// ESP32 Soundboard — Core Firmware // Install: DFRobotDFPlayerMini library #include <Arduino.h> #include <DFRobotDFPlayerMini.h> #include <WiFi.h> #include <ESPAsyncWebServer.h> // --- WiFi credentials --- const char* ssid = "YOUR_WIFI_NAME"; const char* password = "YOUR_WIFI_PASSWORD"; // --- Button pins --- const int BUTTONS[] = {18, 19, 21, 22, 23}; const int NUM_BUTTONS = 5; const int BANK_UP_PIN = 25; const int BANK_DOWN_PIN = 26; // --- DFPlayer on UART2 --- HardwareSerial dfSerial(2); DFRobotDFPlayerMini dfPlayer; AsyncWebServer server(80); // --- State --- int currentBank = 0; int numBanks = 3; bool lastButtonState[] = {false,false,false,false,false}; void setup() { Serial.begin(115200); // Init buttons for (int i = 0; i < NUM_BUTTONS; i++) { pinMode(BUTTONS[i], INPUT); } pinMode(BANK_UP_PIN, INPUT); pinMode(BANK_DOWN_PIN, INPUT); // Init DFPlayer dfSerial.begin(9600, SERIAL_8N1, 16, 17); dfPlayer.begin(dfSerial); dfPlayer.volume(20); // Connect WiFi WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); } Serial.println(WiFi.localIP()); // Print IP to Serial Monitor // Web server routes (expanded in full code) server.begin(); } void loop() { for (int i = 0; i < NUM_BUTTONS; i++) { bool pressed = digitalRead(BUTTONS[i]) == HIGH; if (pressed && !lastButtonState[i]) { // File number = bank * 10 + button + 1 // Bank 0 btn 0 = file 1, Bank 1 btn 0 = file 11, etc. int fileNum = (currentBank * NUM_BUTTONS) + i + 1; dfPlayer.play(fileNum); } lastButtonState[i] = pressed; } // Bank switching if (digitalRead(BANK_UP_PIN) == HIGH) currentBank = (currentBank + 1) % numBanks; if (digitalRead(BANK_DOWN_PIN) == HIGH) currentBank = (currentBank - 1 + numBanks) % numBanks; delay(50); // Debounce }

3. SD Card File Naming

DFPlayer reads files by number. Name your MP3s like this:

Bank 0 (Daily Needs): 0001.mp3 → Button 1 "I need to go potty" 0002.mp3 → Button 2 "Change the show please" 0003.mp3 → Button 3 "I'm hungry" 0004.mp3 → Button 4 "I want a hug" 0005.mp3 → Button 5 "I'm tired" Bank 1 (TV Control): 0006.mp3 → Button 1 "Turn it up" 0007.mp3 → Button 2 "Turn it down" ... and so on
✓ Free TTS tools to make the audio files: Google Text-to-Speech, ElevenLabs (free tier), or just record yourself on your phone. Export as MP3, rename the file, drop on SD card.
